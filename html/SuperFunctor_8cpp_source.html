26 SuperFunctor::operator EntryPointFP()
27 {
28  if(!entrypoint_fpt)
29  {
30  // First fill in the FPT (Function Pointer with This arg) using
31  // hand-written version of the vcall algorithm. We do it here
32  // because the object is not fully formed during our constructor.
33  EntryPointMFP mfp = &SuperFunctor::operator();
34 
35  // alias an array of words over the member function pointer
36  typedef array<uint32_t, sizeof(EntryPointMFP)/sizeof(uint32_t)> MFPWords;
37  MFPWords &mfp_words = *(MFPWords *)&mfp;
38  ASSERT( mfp_words.size() == member_function_pointer_size/4, "Unexpected size of member function pointer");
39 
40  auto this_words = (uint32_t *)this;
41  entrypoint_fpt = (EntryPointFPT)get_vcall_destination( mfp_words.data(), this_words );
42  }
43 
44  // Convert the address of the trampoline to a function pointer
45  return (EntryPointFP)ptr_to_function_ptr(entrypoint_trampoline);
46 }

49 pair<MachineInstruction *, int> SuperFunctor::GetAssembly()
50 {
51  const int pipeline_overshoot = pipeline_overshoot_instructions * sizeof(MachineInstruction);
52 
53  volatile bool actually_run_it = false;
54  if( actually_run_it )
55  {
56  BEGIN:
57  asm( SUPER_FUNCTOR_TRAMPOLINE
58  : : [this_to_PC_offset] "I" (offsetof(SuperFunctor, entrypoint_trampoline) + pipeline_overshoot),
59  [this_to_entrypoint_offset] "M" (offsetof(SuperFunctor, entrypoint_fpt)) : );
60  END:
61  for(;;);
62  }
63 
64  auto begin = (MachineInstruction *)(&&BEGIN);
65  auto end = (MachineInstruction *)(&&END);
66 
67  return make_pair( begin, end-begin );
68 }
