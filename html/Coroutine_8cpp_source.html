27 Coroutine::Coroutine( function<void()> child_function_ ) :
28  child_function( child_function_ ),
29  stack_size( default_stack_size ),
30  child_stack_memory( (byte *)calloc(default_stack_size, 1) ),
31  child_status( READY )
32 {
33  ASSERT(child_function, "NULL child function was supplied");
34  jmp_buf initial_jmp_buf;
35  int val;
36  CONSTRUCTOR_TRACE("this=%p sp=%p", this, get_sp());
37  CONSTRUCTOR_TRACE("last arg begins %p ends %p", &child_function_, &child_function_ + 1);
38  CONSTRUCTOR_TRACE("first local begins %p ends %p", &initial_jmp_buf, &initial_jmp_buf + 1);
39  switch( val = setjmp(initial_jmp_buf) ) {
40  case IMMEDIATE: {
41  // Get current stack pointer and frame address
42  // taking care that it will have a different stack frame
43  byte *frame_end = (byte *)(&child_function_ + 1);
44  byte *stack_pointer = (byte *)( get_jmp_buf_sp(initial_jmp_buf) );
45 
46  CONSTRUCTOR_TRACE("this=%p sp=%p, stack_pointer=%p, frame_end=%p", this, get_sp(), stack_pointer, frame_end);
47 
48  // Get the child's stack ready
49  byte *child_stack_pointer = prepare_child_stack( frame_end, stack_pointer );
50 
51  // Create the child's jump buf
52  prepare_child_jmp_buf( child_jmp_buf, initial_jmp_buf, stack_pointer, child_stack_pointer );
53  break;
54  }
55  case PARENT_TO_CHILD_STARTING: {
56  CONSTRUCTOR_TRACE("this=%p me=%p sp=%p", this, me(), get_sp());
57  child_main_function();
58  }
59  default: {
60  // This setjmp call was only to get the stack pointer.
61  ERROR("unexpected longjmp value: %d", val);
62  }
63  }
64 }

117 byte *Coroutine::prepare_child_stack( byte *frame_end, byte *stack_pointer )
118 {
119  // Decide how much stack to keep (basically the current frame, i.e. the
120  // stack frame of this invocation of this function) and copy it into the
121  // new stack, at the bottom.
122  // Note: stacks usually begin at the highest address and work down
123  int bytes_to_retain = frame_end - stack_pointer;
124  byte *child_stack_pointer = child_stack_memory + stack_size - bytes_to_retain;
125  CONSTRUCTOR_TRACE("moving %d from %p to %p", bytes_to_retain, stack_pointer, child_stack_pointer );
126  memmove( child_stack_pointer, stack_pointer, bytes_to_retain );
127  return child_stack_pointer;
128 }

137  byte *parent_frame_pointer = (byte *)(get_jmp_buf_fp(initial_jmp_buf));

231 void *Coroutine::get_cls_address(void *obj)
232 {
233  __emutls_object * const euo = (__emutls_object *)obj;
234  const Coroutine *me = ::me();
235  if( euo->loc.offset==0 )
236  {
237  // The first time a TLS item is accessed, regardless of context, we
238  // give it an offset.
239  if( cls_heap_top==0 )
240  cls_heap_top++; // If we put 0 into euo.loc.offset, it will be indistinguishable from nullptr
241  euo->loc.offset = (cls_heap_top + euo->align - 1) & ~(euo->align - 1);
242  cls_heap_top = euo->loc.offset + euo->size;
243  }
244 
245  byte *cls_heap;
246  if( me )
247  {
248  // TLS data accessed in a coroutine
249  cls_heap = me->child_stack_memory;
250  }
251  else
252  {
253  // TLS data accessed outside of any coroutine
254  // The first time this happens, we'll have to allocate a block of memory
255  if( !cls_foreground_heap )
256  cls_foreground_heap = (byte *)calloc(default_stack_size, 1);
257  cls_heap = cls_foreground_heap;
258  }
259  return cls_heap + euo->loc.offset;
260 }
